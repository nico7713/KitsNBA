CREATE TABLE "usuarios" (
	"id_usuario"	INTEGER,
	"nombre"	TEXT,
	"apellido"	TEXT,
	"num_telefono"	TEXT,
	"email"	TEXT,
	"id_ubicacion"	INTEGER,
	"username"	TEXT,
	"password"	INTEGER,
	"tipo_usuario"	TEXT,
	PRIMARY KEY("id_usuario" AUTOINCREMENT)
);

);


CREATE TABLE "stock" (
	"id_stock"	INTEGER,

	"stock"	INTEGER,
	PRIMARY KEY("id_stock" AUTOINCREMENT)
);

	"direccion"	TEXT,

);

	"id_producto"	INTEGER,

	"jugador"	TEXT,
	"verion"	TEXT,

);
